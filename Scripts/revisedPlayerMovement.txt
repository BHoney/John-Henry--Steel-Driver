show_debug_message("Step");
//Check player input

move_left = keyboard_check(vk_left);
move_right = keyboard_check(vk_right);
move_jump = keyboard_check_pressed(vk_space);

//Read the Inputs
if !place_meeting(x, y, par_terrain)
   {
   move_contact_solid(270, gravity);
   show_debug_message("Falling");
   }
 

sprite_index = John;
    if (move_left)
    {
        image_xscale = -1;
        sprite_index = JhenyWalk;
        xSpeed = -1 ;
        show_debug_message("Moving left");
    }
    
    else if (move_right)
    {
        image_xscale = 1;
        sprite_index = JhenyWalk;
        xSpeed = 1;
        show_debug_message("Moving Right");
    }
    
    else
    {
        sprite_index = John;
        xSpeed = 0;
    }
    
    if (move_jump && ySpeed == 0)
    {
        ySpeed = -5;
        show_debug_message("Jumping");
    }      
    
   
    direction = point_direction(x, y, x+xSpeed, y+ySpeed);
    show_debug_message(direction);
    speed = sqrt(xSpeed * xSpeed + ySpeed * ySpeed);
    show_debug_message(speed);
    
    if(ySpeed > 12)
    {
         ySpeed = 12;
         show_debug_message("Terminal Velocity Reached");
    }
    if (place_meeting(x, y+1, blockFloor) && ySpeed > 0)
    {
        ySpeed = 0;
        show_debug_message("grounded");
    }

    ySpeed += grav;
